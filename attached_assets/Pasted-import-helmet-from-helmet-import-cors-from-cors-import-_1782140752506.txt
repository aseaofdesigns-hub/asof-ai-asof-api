// --- Payment auth middleware ---
// This is the critical one: blocks /api/validate without a paid session
export function requirePaidSession(req: Request, res: Response, next: NextFunction) {
  // Option A: Check a session token/cookie set after Stripe checkout completes
  const sessionToken = req.headers.authorization?.replace("Bearer ", "") 
    || req.cookies?.asofai_session;

  if (!sessionToken) {
    return res.status(401).json({ 
      error: "Payment required. Please complete checkout first." 
    });
  }

  // Validate the token against your database
  // Replace this with your actual session/payment lookup
  validatePaymentSession(sessionToken)
    .then((session) => {
      if (!session || session.used) {
        return res.status(403).json({ 
          error: "Session expired or already used." 
        });
      }
      // Attach session to request for downstream use
      (req as any).paidSession = session;
      next();
    })
    .catch(() => {
      return res.status(500).json({ error: "Session validation failed." });
    });
}

// --- 404 for unknown API routes (put AFTER all real API routes, BEFORE SPA catch-all) ---
export function apiNotFound(req: Request, res: Response, next: NextFunction) {
  if (req.path.startsWith("/api/")) {
    return res.status(404).json({ error: "Endpoint not found" });
  }
  next();
}
