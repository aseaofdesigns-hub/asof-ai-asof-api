export type Tier = "lite" | "pro" | "max";

export const TIER_FEATURES: Record<Tier, { label: string; locked: boolean }[]> = {
  lite: [
    { label: "SUMMARY", locked: false },
    { label: "ASSUMPTIONS", locked: false },
    { label: "WHAT COULD BREAK", locked: false },
    { label: "VERIFY CHECKLIST", locked: true },
    { label: "SUGGESTIONS", locked: true },
    { label: "SAFE REWRITES", locked: true },
  ],
  pro: [
    { label: "SUMMARY", locked: false },
    { label: "ASSUMPTIONS", locked: false },
    { label: "WHAT COULD BREAK", locked: false },
    { label: "VERIFY CHECKLIST", locked: false },
    { label: "SUGGESTIONS", locked: false },
    { label: "SAFE REWRITES", locked: true },
  ],
  max: [
    { label: "SUMMARY", locked: false },
    { label: "ASSUMPTIONS", locked: false },
    { label: "WHAT COULD BREAK", locked: false },
    { label: "VERIFY CHECKLIST", locked: false },
    { label: "SUGGESTIONS", locked: false },
    { label: "SAFE REWRITES", locked: false },
  ],
};

export const TIER_ORDER: Tier[] = ["lite", "pro", "max"];

export const UPGRADE_PRICES: Record<string, number> = {
  "lite->pro": 0.50,
  "lite->max": 2.00,
  "pro->max": 2.00,
};

export function getNextUpgrade(currentTier: Tier): Tier | null {
  const idx = TIER_ORDER.indexOf(currentTier);
  if (idx >= TIER_ORDER.length - 1) return null;
  return TIER_ORDER[idx + 1];
}

export function getAvailableUpgrades(currentTier: Tier): Tier[] {
  const idx = TIER_ORDER.indexOf(currentTier);
  return TIER_ORDER.slice(idx + 1);
}

**4. Updated upgrade box** — `client/components/UpgradeBox.tsx`

```tsx
import { getNextUpgrade, UPGRADE_PRICES, type Tier } from "../lib/tierFeatures";

interface UpgradeBoxProps {
  currentTier: Tier;
  originalTier: Tier; // what they originally paid for
  onUpgrade: (toTier: Tier) => void;
}

export function UpgradeBox({ currentTier, originalTier, onUpgrade }: UpgradeBoxProps) {
  const nextTier = getNextUpgrade(currentTier);

  // Already at max — no upgrade available
  if (!nextTier) return null;

  const priceKey = `${originalTier}->${nextTier}` as keyof typeof UPGRADE_PRICES;
  const price = UPGRADE_PRICES[priceKey] ?? UPGRADE_PRICES[`${currentTier}->${nextTier}`];

  const tierLabels: Record<Tier, string> = {
    lite: "Lite",
    pro: "Pro — Verify checklist + suggestion cards",
    max: "Max — Safer code rewrite side-by-side",
  };

  return (
    <div className="rounded-lg border border-white/10 bg-white/5 p-4">
      <p className="text-sm font-semibold text-white mb-3 flex items-center gap-2">
        <span>⬆</span> Upgrade this analysis — pay only the difference
      </p>
      <button
        onClick={() => onUpgrade(nextTier)}
        className="w-full flex items-center justify-between rounded-lg border border-green-500/30 bg-green-500/10 px-4 py-3 text-sm text-white hover:bg-green-500/20 transition-colors"
      >
        <span className="flex items-center gap-2">
          <span className="text-green-400">⊕</span>
          {tierLabels[nextTier]}
        </span>
        <span className="text-green-400 font-bold">+${price.toFixed(2)}</span>
      </button>
      <p className="text-xs text-white/30 mt-2 text-center">
        Same analysis, instantly expanded — no re-run needed.
      </p>
    </div>
  );
}
```
